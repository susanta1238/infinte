# InfiniteTalk + API server consolidated requirements
# Verified against PyPI on 2026-04-22.
#
# Target pod image: runpod/pytorch:2.4.0-py3.11-cuda12.4.1-devel-ubuntu22.04
#   - ships torch 2.4.0 built for CUDA 12.4, Python 3.11
#   - includes nvcc + headers (devel) so flash-attn can build if no wheel
#
# IMPORTANT: install torch, xformers, and flash_attn on the pod BEFORE running
# `pip install -r requirements.txt`. The base image already has torch 2.4.0+cu124
# installed, so normally you only need:
#
#   pip install xformers==0.0.27.post2
#   pip install flash_attn==2.7.4.post1
#
# WHY 0.0.27.post2 (not 0.0.28.*): every xformers 0.0.28.postN PyPI wheel
# (post1/post2/post3) was built against torch 2.5.x. Running them on torch 2.4
# silently disables the C++ kernels (verified: "xFormers was built for PyTorch
# 2.5.x (you have 2.4.1+cu124)" warning at import, then every call to
# memory_efficient_attention raises "No operator found").
# 0.0.27.post2 is the last release compiled against torch 2.4.
#
# The pins below exist so that resolving this file CANNOT upgrade torch or its
# CUDA-linked siblings. Without these pins, `accelerate>=1.1.1` resolves to
# accelerate 1.13.x which declares `torch>=2.0.0` with no upper bound -- pip
# then installs torch 2.11 (CUDA 13), which breaks flash_attn 2.7.4.post1 and
# xformers because they were compiled against the torch 2.4 ABI.

# --- Hard pins: DO NOT upgrade these via requirements.txt ---
# torch pinned to the version shipped in the RunPod base image.
torch==2.4.0
torchvision==0.19.0
torchaudio==2.4.0
# xformers: MUST match torch 2.4 ABI.
# IMPORTANT FINDING: all xformers 0.0.28.postN PyPI wheels (post1/post2/post3)
# were built against torch 2.5.x -- the C++ extension fails to load on torch 2.4,
# xformers.ops falls back to slow/no-op path, and the audio cross-attention
# in wan/modules/attention.py can't run. Pin to 0.0.27.post2 which is the last
# release explicitly compiled against torch 2.4.
xformers==0.0.27.post2
flash_attn==2.7.4.post1

# --- Core ML / diffusion stack ---
# transformers v5 introduces breaking changes; pin to 4.x until upstream model code is ported.
# transformers UPPER bound is load-bearing:
# 4.50 dropped the Flax-related re-exports (FLAX_WEIGHTS_NAME etc.) from
# transformers.utils. diffusers 0.32 still imports FLAX_WEIGHTS_NAME at module
# load time, so any transformers>=4.50 breaks `from diffusers import DiffusionPipeline`
# before a single pipeline can be instantiated.
transformers>=4.49.0,<4.50
tokenizers>=0.20.3,<0.23
# diffusers: UPPER bound is load-bearing. Versions >=0.33 register a
# torch custom op using PEP-604 `torch.Tensor | None` syntax in the
# attention-dispatch module. torch 2.4's `_library.infer_schema` does
# not understand that syntax and raises "Parameter q has unsupported
# type torch.Tensor" at import time, killing any pipeline load.
# diffusers 0.31.0 pre-dates that module and works on torch 2.4.
diffusers>=0.31.0,<0.33
# Cap accelerate so it keeps a torch<2.5 compatible declaration.
# (accelerate 1.3 was the last release before the open torch bound.)
accelerate>=1.1.1,<1.3
optimum-quanto>=0.2.7

# --- Numerical / image / video ---
# numpy 2.x breaks several pinned C-extensions in this stack (xfuser, decord fork). Keep <2.
numpy>=1.23.5,<2
opencv-python>=4.9.0.80
scikit-image
imageio
imageio-ffmpeg
# Original `decord` (PyPI) is unmaintained since 2021. eva-decord is a drop-in fork
# that preserves the `import decord` API used in wan/utils/{utils,vace_processor,qwen_vl_utils}.py.
eva-decord>=0.6.1
# moviepy is pinned to 1.0.3 because wan/utils/segvideo.py uses `from moviepy.editor import *`,
# which was removed in moviepy 2.x. Do not bump without porting the import.
moviepy==1.0.3
scenedetect>=0.6.7

# --- Distributed inference ---
# xfuser UPPER bound is load-bearing:
# xfuser >=0.4.5 imports `SanaAttnProcessor2_0` from
# diffusers.models.transformers.sana_transformer at module load time.
# That symbol was introduced in diffusers 0.33. We can't use diffusers>=0.33
# (the attention-dispatch module there uses PEP-604 custom-op syntax that
# torch 2.4's _library.infer_schema rejects -- see diffusers pin above).
# So we downgrade xfuser to a release that works with diffusers 0.32.
xfuser>=0.4.3,<0.4.5

# --- Audio ---
# Used by generate_infinitetalk.py (audio_prepare_*, get_embedding) and app.py.
librosa
soundfile
pyloudnorm
einops

# --- UI / CLI / utilities ---
gradio>=5.0.0
tqdm
loguru
ftfy
easydict
dashscope
safetensors>=0.4.3

# --- API server (was api_requirements.txt) ---
fastapi>=0.115
uvicorn[standard]>=0.30
# 0.0.18+ fixes CVE-2024-53981 (DoS via malformed multipart).
python-multipart>=0.0.18
